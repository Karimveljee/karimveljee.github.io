x <- seq(-10, 10, length=30)

# Define the function for z values
f <- function(x, y) {
 r <- sqrt(x^2 + y^2)
 z <- sin(r) / r
 z[is.na(z)] <- 1 # Handle NaN values
 return(z)
}
# Create the z matrix
z <- outer(x, y, f)
# Create the 3D perspective plot
